//a helper function to make a recipe card function makeRecipeCard(recipe) { var card = document.createElement("li"); var cardLink = document.createElement("a"); var cardTitle = document.createElement("h2"); var cardImage = document.createElement("img"); var cardAvatar = document.createElement("img"); cardLink.href = "/cookbook?id=" + recipe._id; cardLink.classList.add("wrapper"); cardTitle.textContent = recipe.title; cardImage.src = recipe.image || "/asset/FFFFFF-0.png"; cardImage.setAttribute("loading", "lazy"); cardImage.classList.add("skeletoned"); //if an image is defined, un-skeleton it when it loads cardImage.addEventListener("load", function() { if(recipe.image) cardImage.classList.remove("skeletoned"); }); var cardByline = makeCardByline(recipe.ownerUserId); cardLink.appendChild(cardTitle); cardLink.appendChild(cardByline); cardLink.appendChild(cardImage); card.appendChild(cardLink); return card; }
